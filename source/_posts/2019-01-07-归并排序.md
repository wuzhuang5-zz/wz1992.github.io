采用分治思想，时间复杂度为O(nlogn)且稳定的排序,空间复杂度为O(n)，可以实现原地排序

```java
public class MergeSort {

    public static void mergeSort(int[] nums) {
        mergeSortInner(nums, 0, nums.length - 1);
    }
	//递归调用分解数组
    public static void mergeSortInner(int[] nums, int l, int r) {
        if (l >= r) {
            return;
        }
        int q = l + (r - l) / 2;
        mergeSortInner(nums, l, q);
        mergeSortInner(nums, q + 1, r);
        merge(nums, l, q, r);
    }
	//合并数组
    public static void merge(int[] nums, int l, int q, int r) {
        int i = l;
        int j = q + 1;
        int k = 0;
        int[] arr = new int[r - l + 1];
        while (i <= q && j <= r) {
            if (nums[i] <= nums[j]) {
                arr[k++] = nums[i++];
            } else {
                arr[k++] = nums[j++];
            }
        }
        //将剩余的数拷贝到临时数组中
        int start = i;
        int end = q;
        if (j <= r) {
            start = j;
            end = r;
        }
        while (start <= end) {
            arr[k++] = nums[start++];
        }
        //将临时数组拷贝回原数组
        for (i = 0; i <= r - l; i++) {
            nums[l+i] = arr[i];
        }
    }
}
```
